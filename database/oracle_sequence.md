# 시퀀스(Sequence)

```sql
--시퀀스 : 칼럼 값 순차적으로 증가시키는 객체

--하나의 시퀀스로 여러 테이블 동시적용 가능(하나는 NEXTVAL, 다른 테이블 CURRVAL)



DROP SEQUENCE MEMBER_NO_SEQ;

CREATE SEQUENCE MEMBER_NO_SEQ
INCREMENT BY 1 --증가하는 값 설정 가능
START WITH 1 --시작값
MINVALUE 1 --최소값
MAXVALUE 999 --최대값
NOCYCLE --순환여부
NOCACHE -- 캐시 여부(사이즈) --NO CACHE가 안전하다
;

SELECT MEMBER_NO_SEQ.NEXTVAL FROM DUAL;
SELECT MEMBER_NO_SEQ.CURRVAL FROM DUAL;

--사용
SELECT MEMBER_NO_SEQ.NEXTVAL --CURRVAL : 현재의 시퀀스값(NEXTVAL을 한 번은 해줘야함)
FROM DUAL;

--수정(START WITH 외에는 모두 변경가능)
ALTER SEQUENCE MEMBER_NO_SEQ
INCREMENT BY 2
MINVALUE -10
MAXVALUE 123
CYCLE
CACHE 50
;

CREATE SEQUENCE ASDF_NO_SEQ
INCREMENT BY 2
START WITH 1
MINVALUE 1
MAXVALUE 1
CYCLE
CACHE 50
;

ALTER SEQUENCE ASDF_NO_SEQ
INCREMENT BY 3
MINVALUE 2
MAXVALUE 10
NOCYCLE
CACHE 50

```
